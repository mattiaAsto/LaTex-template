\begin{figure}[h]
  \centering
  \begin{tikzpicture}
    \begin{axis}[
      title={Function Plot},
      xlabel={x},
      ylabel={f(x)},
      xmin=-5, xmax=5,
      ymin=-1, ymax=1,
      axis lines=middle,
      grid=major,
      width=0.8\textwidth,
      height=6cm,
      trig format plots=rad
    ]
    
    \addplot[
      name path=A,
      domain=-5:5,
      samples=100,
      color=red,
      smooth
    ] {sin(x)};
    \addlegendentry{ciao}

  %Remember to add 'name path=<...>' in your \addplot, as first argument
  %Next line is to use x axes as base for filling
  \path[name path=B] (axis cs:\pgfkeysvalueof{/pgfplots/xmin},0) -- (axis cs:\pgfkeysvalueof{/pgfplots/xmax},0);
  %line used to create te filling
  \addplot[color=yellow!30!red, opacity=0.3] fill between [
  of=A and B,
  soft clip={domain=-0.5:0.5} % Qui definisci i due valori (X1:X2)
  ];
    
    
  \end{axis}
  \end{tikzpicture}
  \caption{Plot of the function $x^2$}
  \label{fcn2d:fcn_plot}
\end{figure}
